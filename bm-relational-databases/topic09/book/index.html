Logical Comparisons and Precedence Rules

SELECT code, description
FROM d_themes
WHERE code > 200 AND description IN ('Tropical', 'Football', 'Carnival');

SELECT code, description
FROM d_themes
WHERE code > 200 OR description IN ('Tropical', 'Football','Carnival');
